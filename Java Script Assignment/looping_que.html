Q32: Write to print factorial of given number?

Ans : 

 
            let num = prompt('enter a number');

            function fac(num){
                let answer = 1;
                if (num == 0 || num == 1){
                  return answer;
                }
                else if(num > 1){
                  for(var i = num; i >= 1; i--){
                    answer = answer * i;
                  }
                  return answer;
                }
                else{
                  return "number has to be positive."
                }  
              }
              
              answer = fac(num)
              console.log(answer);

Q34: Write to print number in reverse order e.g.: number = 64728 ---> reverse =82746 in JS?

Ans : 

 
            let num = 123456789;
            
            let rev = 0;
            while(num>0){

                last_num = num%10;
                
                rev = rev*10 + last_num;
                
                num = Math.floor(num/10);
            }
            console.log(rev);

Q36: Write a program you have to make a summation of first and last Digit. (E.g., 1234 Ans: -5) in JS?

            function fdigit(n) 
            { 
                while (n >= 10)  
                    n /= 10; 
                
                return Math.floor(n); 
            } 
            
            function ldigit(n) 
            { 
                return Math.floor(n % 10); 
            }
